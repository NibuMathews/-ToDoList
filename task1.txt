class ToDoList:
    def __init__(self):
        self.tasks = []

    def add_task(self, task):
        self.tasks.append({'task': task, 'completed': False})
        print(f"Added task: {task}")

    def view_tasks(self):
        if not self.tasks:
            print("No tasks in the list.")
            return
        for i, task in enumerate(self.tasks, start=1):
            status = "Completed" if task['completed'] else "Pending"
            print(f"{i}. {task['task']} - {status}")

    def mark_complete(self, task_index):
        try:
            self.tasks[task_index - 1]['completed'] = True
            print(f"Marked task {task_index} as complete.")
        except IndexError:
            print("Invalid task number.")

    def delete_task(self, task_index):
        try:
            task = self.tasks.pop(task_index - 1)
            print(f"Deleted task: {task['task']}")
        except IndexError:
            print("Invalid task number.")

    def update_task(self, task_index, new_task):
        try:
            self.tasks[task_index - 1]['task'] = new_task
            print(f"Updated task {task_index} to: {new_task}")
        except IndexError:
            print("Invalid task number.")

def main():
    todo_list = ToDoList()
    while True:
        print("\nTo-Do List Application")
        print("1. Add a new task")
        print("2. View all tasks")
        print("3. Update a task")
        print("4. Delete a task")
        print("5. Mark a task as complete")
        print("6. Exit")
        choice = input("Enter your choice: ")

        if choice == '1':
            task = input("Enter the task: ")
            todo_list.add_task(task)
        elif choice == '2':
            todo_list.view_tasks()
        elif choice == '3':
            task_index = int(input("Enter the task number to update: "))
            new_task = input("Enter the new task: ")
            todo_list.update_task(task_index, new_task)
        elif choice == '4':
            task_index = int(input("Enter the task number to delete: "))
            todo_list.delete_task(task_index)
        elif choice == '5':
            task_index = int(input("Enter the task number to mark as complete: "))
            todo_list.mark_complete(task_index)
        elif choice == '6':
            print("Exiting the application. Goodbye!")
            break
        else:
            print("Invalid choice. Please try again.")

if __name__ == "__main__":
    main()
